281uint16_t LoRaSlave::readBatteryMilliVolts() const
282{
283 uint32_t acc = 0;
284
285#ifdef BOARD_T114
286 // T114 : activer la broche enable, attendre la stabilisation
287 pinMode(PIN_BAT_ADC_EN, OUTPUT);
288 digitalWrite(PIN_BAT_ADC_EN, HIGH);
289 delay(20);
290 pinMode(PIN_BAT_ADC, INPUT);
291
292 // Echantillons de chauffe (rejetes)
293 for (uint8_t i = 0; i < BAT_WARMUP_COUNT; i++)
294 {
295 watchdogFeed();
296 (void)analogRead(PIN_BAT_ADC);
297 delay(BAT_WARMUP_DELAY_MS);
298 }
299
300 for (uint8_t i = 0; i < BAT_SAMPLE_COUNT; i++)
301 {
302 watchdogFeed();
303 acc += analogRead(PIN_BAT_ADC);
304 delay(BAT_SAMPLE_DELAY_MS);
305 }
306
307 digitalWrite(PIN_BAT_ADC_EN, LOW);
308
309#else // BOARD_RAK4631
310 // RAK4631 : lecture directe sur PIN_VBAT (ou A0 en fallback)
311#if defined(PIN_VBAT)
312 const uint32_t vbat_pin = PIN_VBAT;
313#else
314 const uint32_t vbat_pin = A0;
315#endif
316
317 for (uint8_t i = 0; i < BAT_SAMPLE_COUNT; i++)
318 {
319 watchdogFeed();
320 acc += analogRead(vbat_pin);
321 delay(BAT_SAMPLE_DELAY_MS);
322 }
323#endif
324
325 float raw = (float)acc / BAT_SAMPLE_COUNT;
326 float mv = raw * REAL_VBAT_MV_PER_LSB;
327
328 if (mv < 0.0f)
329 {
330 mv = 0.0f;
331 }
332 if (mv > 65535.0f)
333 {
334 mv = 65535.0f;
335 }
336
337 return (uint16_t)(mv + 0.5f);
338}

343uint16_t LoRaSlave::crc16Ccitt(const uint8_t *data, uint16_t len) const
344{
345 uint16_t crc = 0xFFFF;
346
347 for (uint16_t i = 0; i < len; i++)
348 {
349 crc ^= (uint16_t)data[i] << 8;
350
351 for (uint8_t b = 0; b < 8; b++)
352 {
353 if (crc & 0x8000)
354 crc = (uint16_t)((crc << 1) ^ 0x1021);
355 else
356 crc = (uint16_t)(crc << 1);
357 }
358 }
359
360 return crc;
361}

368uint8_t LoRaSlave::buildFrame(uint8_t type, uint8_t nodeId, uint8_t seq,
369 const uint8_t *payload, uint8_t payloadLen)
370{
371 txBuffer[0] = type;
372 txBuffer[1] = nodeId;
373 txBuffer[2] = seq;
374 txBuffer[3] = payloadLen;
375
376 for (uint8_t i = 0; i < payloadLen; i++)
377 {
378 txBuffer[4 + i] = payload[i];
379 }
380
381 uint16_t crc = crc16Ccitt(txBuffer, (uint16_t)(4 + payloadLen));
382 txBuffer[4 + payloadLen] = (uint8_t)(crc & 0xFF);
383 txBuffer[5 + payloadLen] = (uint8_t)((crc >> 8) & 0xFF);
384
385 return getFrameTotalLength(payloadLen);
386}

388bool LoRaSlave::validateFrame(uint8_t expectedType, uint8_t expectedNode,
389 uint8_t expectedSeq) const
390{
391 if (rxSize < 6)
392 {
393 DBG.println("ERR frame too short");
394 return false;
395 }
396 if (rxBuffer[0] != expectedType)
397 {
398 DBG.println("ERR frame type");
399 return false;
400 }
401 if ((rxBuffer[1] != expectedNode) && (rxBuffer[1] != 0xFF))
402 {
403 DBG.println("ERR node id");
404 return false;
405 }
406 if (rxBuffer[2] != expectedSeq)
407 {
408 DBG.println("ERR seq");
409 return false;
410 }
411
412 uint8_t payloadLen = rxBuffer[3];
413 uint16_t expectedTotal = (uint16_t)(4 + payloadLen + 2);
414
415 if (expectedTotal > MAX_BUFFER_SIZE)
416 {
417 DBG.println("ERR frame oversize");
418 return false;
419 }
420 if (rxSize != expectedTotal)
421 {
422 DBG.println("ERR frame length");
423 return false;
424 }
425
426 uint16_t rxCrc = (uint16_t)rxBuffer[4 + payloadLen] |
427 ((uint16_t)rxBuffer[5 + payloadLen] << 8);
428 uint16_t calcCrc = crc16Ccitt(rxBuffer, (uint16_t)(4 + payloadLen));
429
430 if (rxCrc != calcCrc)
431 {
432 DBG.print("ERR CRC rx=");
433 DBG.print(rxCrc, HEX);
434 DBG.print(" calc=");
435 DBG.println(calcCrc, HEX);
436 return false;
437 }
438
439 return true;
440}

614 }

619// ============================================================================
620// Protocole HELLO / ACK
621// ============================================================================
622bool LoRaSlave::sendHelloAndWaitAckOnce()
623{
624 uint8_t seq = nextSeq();
625
626 DBG.print("[PROTO] sendHelloAndWaitAck(), seq=");
627 DBG.println(seq);
628
629 uint8_t helloPayload[1];
630 helloPayload[0] = getHelloFlags();
631
632 uint8_t totalLen = buildFrame(FRAME_HELLO, NODE_ID, seq, helloPayload, sizeof(helloPayload));
633
634 if (!sendFrameAndWaitTx(totalLen))
635 {
636 DBG.println("[PROTO] HELLO TX failed");
637 return false;
638 }
639
640 uint32_t deadline = millis() + ACK_WAIT_MS;
641
642 while ((int32_t)(millis() - deadline) < 0)
643 {
644 uint32_t remainingMs = (uint32_t)(deadline - millis());
645 if (remainingMs > 100UL)
646 {
647 remainingMs = 100UL;
648 }
649
650 if (!waitForFrame(remainingMs))
651 {
652 continue;
653 }
654
655 if (rxSize < 6)
656 {
657 DBG.println("[PROTO] Ignore short frame while waiting ACK_HELLO");
658 continue;
659 }
660
661 if (rxBuffer[0] != FRAME_ACK_HELLO)
662 {
663 DBG.print("[PROTO] Ignore non-ACK frame while waiting ACK_HELLO: type=0x");
664 DBG.println(rxBuffer[0], HEX);
665 continue;
666 }
667
668 if (rxBuffer[1] != NODE_ID)
669 {
670 DBG.print("[PROTO] Ignore ACK for other node: ");
671 DBG.println(rxBuffer[1]);
672 continue;
673 }
674
675 if (rxBuffer[2] != seq)
676 {
677 DBG.print("[PROTO] Ignore ACK with wrong seq: ");
678 DBG.println(rxBuffer[2]);
679 continue;
680 }
681
682 if (!validateFrame(FRAME_ACK_HELLO, NODE_ID, seq))
683 {
684 DBG.println("[PROTO] Ignore invalid ACK_HELLO (CRC/len)");
685 continue;
686 }
687
688 _lastAckHelloMs = millis();
689 _justBooted = false;
690 DBG.println("ACK_HELLO OK");
691 return true;
692 }
693
694 DBG.println("[PROTO] No ACK_HELLO");
695 return false;
696}

797// ============================================================================
798// processOptionalCommand() — fenetre multi-trames (implémentation T114)
799// ============================================================================
800bool LoRaSlave::processOptionalCommand()
801{
802 DBG.println("[PROTO] Waiting optional CMD");
803
804 uint32_t startWindowMs = millis();
805
806 while ((millis() - startWindowMs) < CMD_WAIT_MS)
807 {
808 watchdogFeed();
809 watering.loop();
810
811 uint32_t elapsed = millis() - startWindowMs;
812 uint32_t remain = (elapsed < CMD_WAIT_MS) ? (CMD_WAIT_MS - elapsed) : 0;
813
814 if (remain == 0)
815 {
816 break;
817 }
818 if (remain < 50)
819 {
820 remain = 50;
821 }
822
823 if (!waitForFrame(remain))
824 {
825 continue;
826 }
827
828 if (rxSize < 6)
829 {
830 DBG.println("[PROTO] Ignore short frame");
831 continue;
832 }
833
834 uint8_t frameType = rxBuffer[0];
835 uint8_t targetNode = rxBuffer[1];
836 uint8_t seq = rxBuffer[2];
837
838 if (frameType != FRAME_CMD)
839 {
840 DBG.print("[PROTO] Ignore non-CMD frame type=0x");
841 if (frameType < 16)
842 {
843 DBG.print('0');
844 }
845 DBG.println(frameType, HEX);
846 continue;
847 }
848
849 if ((targetNode != NODE_ID) && (targetNode != 0xFF))
850 {
851 DBG.print("[PROTO] Ignore CMD for node ");
852 DBG.println(targetNode);
853 continue;
854 }
855
856 if (!validateFrame(FRAME_CMD, NODE_ID, seq))
857 {
858 DBG.println("[PROTO] Invalid CMD for me");
859 continue;
860 }
861
862 uint8_t payloadLen = rxBuffer[3];
863 if (payloadLen < 1)
864 {
865 DBG.println("[PROTO] Ignore empty CMD");
866 continue;
867 }
868
869 uint8_t cmdCode = rxBuffer[4];
870
871 if (seq == _lastCmdSeq)
872 {
873 DBG.print("[PROTO] Duplicate CMD seq=");
874 DBG.println(seq);
875 delay(ACK_TURNAROUND_DELAY_MS);
876 sendAckCmd(seq, cmdCode, ACK_STATUS_OK, nullptr, 0);
877 return true;
878 }
879 _lastCmdSeq = seq;
880
881 DBG.print("[PROTO] CMD=0x");
882 if (cmdCode < 16)
883 {
884 DBG.print('0');
885 }
886 DBG.println(cmdCode, HEX);
887
888 switch (cmdCode)
889 {
890
891 case CMD_PING:
892 delay(ACK_TURNAROUND_DELAY_MS);
893 return sendAckCmd(seq, CMD_PING, ACK_STATUS_OK, nullptr, 0);
894
895 case CMD_LED_ON:
896 ledSet(true);
897 delay(ACK_TURNAROUND_DELAY_MS);
898 return sendAckCmd(seq, CMD_LED_ON, ACK_STATUS_OK, nullptr, 0);
899
900 case CMD_LED_OFF:
901 ledSet(false);
902 delay(ACK_TURNAROUND_DELAY_MS);
903 return sendAckCmd(seq, CMD_LED_OFF, ACK_STATUS_OK, nullptr, 0);
904
905 case CMD_TEXT_MESSAGE:
906 {
907 DBG.print("[PROTO] TEXT: ");
908 for (uint8_t i = 1; i < payloadLen; i++)
909 {
910 DBG.write((char)rxBuffer[4 + i]);
911 }
912 DBG.println();
913 delay(ACK_TURNAROUND_DELAY_MS);
914 return sendAckCmd(seq, CMD_TEXT_MESSAGE, ACK_STATUS_OK, nullptr, 0);
915 }
916
917 case CMD_BATTERY:
918 {
919 uint16_t batteryMv = readBatteryMilliVolts();
920 uint8_t extra[2];
921 extra[0] = (uint8_t)(batteryMv & 0xFF);
922 extra[1] = (uint8_t)((batteryMv >> 8) & 0xFF);
923 delay(ACK_TURNAROUND_DELAY_MS);
924 return sendAckCmd(seq, CMD_BATTERY, ACK_STATUS_OK, extra, sizeof(extra));
925 }
926
927 case CMD_STATUS:
928 {
929 uint16_t batteryMv = readBatteryMilliVolts();
930 uint8_t statusFlag = getStatusFlag();
931 uint32_t uptimeSec = getUptimeSec();
932
933 uint8_t extra[7];
934 extra[0] = (uint8_t)(batteryMv & 0xFF);
935 extra[1] = (uint8_t)((batteryMv >> 8) & 0xFF);
936 extra[2] = statusFlag;
937 extra[3] = (uint8_t)(uptimeSec & 0xFF);
938 extra[4] = (uint8_t)((uptimeSec >> 8) & 0xFF);
939 extra[5] = (uint8_t)((uptimeSec >> 16) & 0xFF);
940 extra[6] = (uint8_t)((uptimeSec >> 24) & 0xFF);
941
942 delay(ACK_TURNAROUND_DELAY_MS);
943 return sendAckCmd(seq, CMD_STATUS, ACK_STATUS_OK, extra, sizeof(extra));
944 }
945
946 case CMD_SET_PERIOD:
947 {
948 if (payloadLen < 3)
949 {
950 delay(ACK_TURNAROUND_DELAY_MS);
951 return sendAckCmd(seq, CMD_SET_PERIOD, ACK_STATUS_ERROR, nullptr, 0);
952 }
953
954 uint16_t seconds = (uint16_t)rxBuffer[5] | ((uint16_t)rxBuffer[6] << 8);
955
956 if (seconds == 0)
957 {
958 delay(ACK_TURNAROUND_DELAY_MS);
959 return sendAckCmd(seq, CMD_SET_PERIOD, ACK_STATUS_ERROR, nullptr, 0);
960 }
961
962 sleepTimeMs = (uint32_t)seconds * 1000UL;
963 DBG.print("[PROTO] New sleep period ms=");
964 DBG.println(sleepTimeMs);
965
966 delay(ACK_TURNAROUND_DELAY_MS);
967 return sendAckCmd(seq, CMD_SET_PERIOD, ACK_STATUS_OK, nullptr, 0);
968 }
969
970 case CMD_REBOOT:
971 {
972 delay(ACK_TURNAROUND_DELAY_MS);
973 sendAckCmd(seq, CMD_REBOOT, ACK_STATUS_OK, nullptr, 0);
974 delay(100);
975 NVIC_SystemReset();
976 return true;
977 }
978
979 case CMD_VALVE_OPEN:
980 {
981 watering.openValve();
982 delay(ACK_TURNAROUND_DELAY_MS);
983 return sendAckCmd(seq, CMD_VALVE_OPEN,
984 watering.isValveOpen() ? ACK_STATUS_OK : ACK_STATUS_ERROR, nullptr, 0);
985 }
986
987 case CMD_VALVE_CLOSE:
988 {
989 bool ok = watering.closeValve();
990 delay(ACK_TURNAROUND_DELAY_MS);
991 return sendAckCmd(seq, CMD_VALVE_CLOSE,
992 ok ? ACK_STATUS_OK : ACK_STATUS_ERROR, nullptr, 0);
993 }
994
995 case CMD_WATER_TIME:
996 {
997 if (payloadLen < 3)
998 {
999 delay(ACK_TURNAROUND_DELAY_MS);
1000 return sendAckCmd(seq, CMD_WATER_TIME, ACK_STATUS_ERROR, nullptr, 0);
1001 }
1002
1003 uint16_t seconds = (uint16_t)rxBuffer[5] | ((uint16_t)rxBuffer[6] << 8);
1004
1005 if (!watering.startWatering(seconds))
1006 {
1007 delay(ACK_TURNAROUND_DELAY_MS);
1008 return sendAckCmd(seq, CMD_WATER_TIME, ACK_STATUS_ERROR, nullptr, 0);
1009 }
1010
1011 uint8_t extra[17];
1012 fillWaterStatusExtra(extra, watering);
1013 delay(ACK_TURNAROUND_DELAY_MS);
1014 return sendAckCmd(seq, CMD_WATER_TIME, ACK_STATUS_OK, extra, sizeof(extra));
1015 }
1016
1017 case CMD_WATER_ABORT:
1018 {
1019 if (!watering.abortWatering())
1020 {
1021 delay(ACK_TURNAROUND_DELAY_MS);
1022 return sendAckCmd(seq, CMD_WATER_ABORT, ACK_STATUS_ERROR, nullptr, 0);
1023 }
1024
1025 uint8_t extra[17];
1026 fillWaterStatusExtra(extra, watering);
1027 delay(ACK_TURNAROUND_DELAY_MS);
1028 return sendAckCmd(seq, CMD_WATER_ABORT, ACK_STATUS_OK, extra, sizeof(extra));
1029 }
1030
1031 case CMD_WATER_STATUS:
1032 {
1033 uint8_t extra[17];
1034 fillWaterStatusExtra(extra, watering);
1035 delay(ACK_TURNAROUND_DELAY_MS);
1036 return sendAckCmd(seq, CMD_WATER_STATUS, ACK_STATUS_OK, extra, sizeof(extra));
1037 }
1038
1039 case CMD_OTA_MODE:
1040 {
1041 delay(ACK_TURNAROUND_DELAY_MS);
1042 sendAckCmd(seq, CMD_OTA_MODE, ACK_STATUS_OK, nullptr, 0);
1043 delay(100);
1044 enterBleOtaMode();
1045 return true;
1046 }
1047
1048 default:
1049 delay(ACK_TURNAROUND_DELAY_MS);
1050 return sendAckCmd(seq, cmdCode, ACK_STATUS_ERROR, nullptr, 0);
1051 }
1052 }
1053
1054 DBG.println("[PROTO] No optional CMD in window");
1055 return false;
1056}
