143void Watering::loop()
144{
145 updateWateringState();
146
147 if (postCloseMonitorActive)
148 {
149 uint32_t now = millis();
150 uint32_t elapsed = now - postCloseMonitorStartMs;
151
152 // First seconds after close are ignored because water flow does not stop instantly.
153 if (!postCloseIgnoreDone && elapsed >= POST_CLOSE_IGNORE_MS)
154 {
155 noInterrupts();
156 postCloseIgnoredPulses = postCloseRawPulses;
157 interrupts();
158
159 postCloseIgnoreDone = true;
160
161 DBG.print("POST_CLOSE ignored pulses = ");
162 DBG.println(postCloseIgnoredPulses);
163 }
164
165 // End of post-close monitoring window.
166 if (elapsed >= POST_CLOSE_MONITOR_MS)
167 {
168 uint32_t raw;
169
170 noInterrupts();
171 raw = postCloseRawPulses;
172 interrupts();
173
174 uint32_t suspiciousPulses = 0;
175
176 if (postCloseIgnoreDone && raw >= postCloseIgnoredPulses)
177 {
178 suspiciousPulses = raw - postCloseIgnoredPulses;
179 }
180
181 noInterrupts();
182 postClosePulses = suspiciousPulses;
183 interrupts();
184
185 if (postClosePulses >= POST_CLOSE_LEAK_MIN_PULSES)
186 {
187 postCloseLeakDetected = true;
188 }
189
190 DBG.print("POST_CLOSE raw pulses = ");
191 DBG.println(raw);
192
193 DBG.print("POST_CLOSE suspicious pulses = ");
194 DBG.println(postClosePulses);
195
196 DBG.print("POST_CLOSE leak = ");
197 DBG.println(postCloseLeakDetected ? 1 : 0);
198
199 postCloseMonitorActive = false;
200
201 // Now we can really stop the flow sensor and return to low power.
202 detachFlowInterrupt();
203 }
204 }
205}

207bool Watering::openValve()
208{
209 if (valveOpen)
210 {
211 return false;
212 }
213
214 // If a new watering starts during a previous post-close monitor window,
215 // cancel that monitoring window.
216 postCloseMonitorActive = false;
217 postCloseIgnoreDone = false;
218
219 enableBoost9V();
220 delay(WATERING_BOOST_SIMULATION_MS);
221 drvOpenPulse();
222
223 // Power flow sensor, configure pullup, and attach interrupt.
224 attachFlowInterrupt();
225
226 disableBoost9V();
227
228 valveOpen = true;
229 return true;
230}

232bool Watering::closeValve()
233{
234 if (!valveOpen && !wateringActive)
235 {
236 return true;
237 }
238
239 enableBoost9V();
240 delay(WATERING_BOOST_SIMULATION_MS);
241 drvClosePulse();
242 disableBoost9V();
243
244 valveOpen = false;
245 wateringActive = false;
246 programmedDurationSec = 0;
247 wateringStartMs = 0;
248
249 // Start post-close monitoring.
250 // Do NOT detach the flow interrupt here.
251 // Flow sensor must remain powered for POST_CLOSE_MONITOR_MS.
252 noInterrupts();
253 postCloseRawPulses = 0;
254 postClosePulses = 0;
255 postCloseIgnoredPulses = 0;
256 interrupts();
257
258 postCloseIgnoreDone = false;
259 postCloseLeakDetected = false;
260 postCloseMonitorActive = true;
261 postCloseMonitorStartMs = millis();
262
263 DBG.println("POST_CLOSE monitor started");
264
265 return true;
266}

268bool Watering::startWatering(uint16_t durationSec)
269{
270 if (durationSec == 0 || wateringActive || valveOpen)
271 {
272 return false;
273 }
274
275 resetFlowCounter();
276
277 programmedDurationSec = durationSec;
278 wateringStartMs = millis();
279 wateringActive = true;
280
281 if (!openValve())
282 {
283 wateringActive = false;
284 programmedDurationSec = 0;
285 wateringStartMs = 0;
286 return false;
287 }
288
289 return true;
290}

447void Watering::attachFlowInterrupt()
448{
449 int irq = digitalPinToInterrupt(WATERING_PIN_FLOW);
450 if (irq >= 0)
451 {
452 nrf_gpio_cfg_output(FLOWPOWER);
453 nrf_gpio_pin_set(FLOWPOWER);
454 delay(5);
455
456 // Enable pullup only when Ve_3V3 is ON.
457 nrf_gpio_cfg_input(WATERING_PIN_FLOW, NRF_GPIO_PIN_PULLUP);
458
459 detachInterrupt(irq);
460 attachInterrupt(irq, Watering::onFlowPulseISR, WATERING_FLOW_INTERRUPT_MODE);
461 }
462}

464void Watering::detachFlowInterrupt()
465{
466 int irq = digitalPinToInterrupt(WATERING_PIN_FLOW);
467 if (irq >= 0)

469 detachInterrupt(irq);
470 }
471
472 // Avoid back-powering the flow sensor rail.

474
475 nrf_gpio_cfg_output(FLOWPOWER);
476 nrf_gpio_pin_clear(FLOWPOWER);
